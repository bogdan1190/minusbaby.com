 46     /**
 47     	 * @lends $.fn.monobombNavigator

 50     /**
 51     	 * Moves so that the first or specified page is the leftmost visible page
 52     	 *
 53     	 * @example $(element).monobombNavigator('moveToPage', 2)
 54     	 *
 55     	 * @param {mixed} [page] Page number (1 indexed) or the actual page itself
 56     	 * @param {bool} [force] Whether to still move even if the nav is closed
 57     	 * @param {function} [callback] Callback
 58     	 *
 59     	 * @returns {object} Jquery object
 60     */
 61 
 62     moveToPage: function(page, force, callback) {
 63       var $nav, data, time;
 64       data = this.data('monobombNavigator');
 65       if (typeof page === 'function') {
 66         callback = page;
 67         page = null;
 68       }
 69       if (typeof force === 'function') {
 70         callback = force;
 71         force = false;
 72       }
 73       if (data.closed && !force) {
 74         return;
 75       }
 76       if (!page) {
 77         page = 1;
 78       }
 79       time = force ? 0 : 'slow';
 80       $nav = private_.getPageObject.call(this, page);
 81       page = private_.getPageNumber.call(this, page);
 82       data.first_page = page;
 83       data.$inner_nav_wrapper.stop().animate({
 84         left: '-' + $nav.position().left + 'px'
 85       }, time, callback);
 86       methods.setButtons.apply(this);
 87       return this;
 88     },
 89     /**
 90     	 * Opens so that the current or specified page is the leftmost visible page
 91     	 *
 92     	 * @example $(element).monobombNavigator('openToPage', 2)
 93     	 *
 94     	 * @param {mixed} [page] Page number (1 indexed) or the actual page itself
 95     	 * @param {function} [callback] Callback
 96     	 *
 97     	 * @returns {object} Jquery object
 98     */
 99 
100     openToPage: function(page, callback) {
101       var data;
102       data = this.data('monobombNavigator');
103       data.$more.fadeOut('slow');
104       this.find('> article').add(this.find('.admin')).stop().animate({
105         opacity: 0
106       }, 'slow');
107       data.$controls_nav.fadeIn('slow');
108       data.$main_nav_wrapper.stop().animate({
109         left: 0
110       }, 'slow');
111       data.closed = false;
112       methods.moveToPage.call(this, page, false, callback);
113       return this;
114     },
115     /**
116     	 * Closes the navigator back to its original spot with the current or specified
117     	 * page staying visible
118     	 *
119     	 * @example $(element).monobombNavigator('closeToPage', 2, false)
120     	 *
121     	 * @param {mixed} [page] Page number (1 indexed) or the actual page itself
122     	 * @param {boolean} [show_articles] Should we redisplay the articles?
123     	 * @param {function} [callback] Callback
124     	 *
125     	 * @returns {object} Jquery object
126     */
127 
128     closeToPage: function(page, show_articles, callback) {
129       var data;
130       data = this.data('monobombNavigator');
131       if (data.$main_nav_wrapper.position().left === data.original_left) {
132         return;
133       }
134       if (typeof page === 'boolean') {
135         callback = fade - articles;
136         show_articles = page;
137         page = null;
138       }
139       if (typeof page === 'function') {
140         callback = page;
141         page = null;
142         show_articles = true;
143       }
144       if (typeof show_articles === 'function') {
145         callback = show_articles;
146         show_articles = true;
147       }
148       methods.moveToPage.call(this, page, callback);
149       data.$controls_nav.fadeOut('slow');
150       data.$more.fadeIn('fast');
151       if (show_articles) {
152         this.find('> article').add(this.find('.admin')).stop().animate({
153           opacity: 100
154         }, 'fast');
155       }
156       data.$main_nav_wrapper.stop().animate({
157         left: data.original_left + 'px'
158       }, 'slow', callback);
159       data.closed = true;
160       return this;
161     },
162     /**
163     	 * Scrolls the navigator one full page back
164     	 *
165     	 * @example $(element).monobombNavigator('back')
166     	 *
167     	 * @param {function} [callback] Callback
168     	 *
169     	 * @returns {object} Jquery object
170     */
171 
172     back: function(callback) {
173       var data;
174       data = this.data('monobombNavigator');
175       if (data.$inner_nav_wrapper.is(':animated')) {
176         return;
177       }
178       methods.moveToPage.call(this, Math.max(1, data.first_page - data.settings.visible_columns));
179       return this;
180     },
181     /**
182     	 * Scrolls the navigator one full page forward
183     	 *
184     	 * @example $(element).monobombNavigator('forward')
185     	 *
186     	 * @param {function} [callback] Callback
187     	 *
188     	 * @returns {object} Jquery object
189     */
190 
191     forward: function(callback) {
192       var data, maxFirstPage, newPage;
193       data = this.data('monobombNavigator');
194       if (data.$inner_nav_wrapper.is(':animated')) {
195         return;
196       }
197       newPage = data.first_page + data.settings.visible_columns;
198       maxFirstPage = data.nav_count - data.settings.visible_columns + 1;
199       if (newPage > maxFirstPage) {
200         newPage = maxFirstPage;
201       }
202       methods.moveToPage.call(this, newPage, false, callback);
203       return this;
204     },
205     /**
206     	 * Sets the control's buttons according to availability of pages
207     	 *
208     	 * @example $(element).monobombNavigator('setButtons')
209     	 *
210     	 * @returns {object} Jquery object
211     */
212 
213     setButtons: function() {
214       var data;
215       data = this.data('monobombNavigator');
216       if (data.nav_count - data.first_page > data.settings.visible_columns - 1) {
217         data.$forward.removeClass('disabled');
218       } else {
219         data.$forward.addClass('disabled');
220       }
221       if (data.first_page > 1) {
222         data.$back.removeClass('disabled');
223       } else {
224         data.$back.addClass('disabled');
225       }
226       return this;
227     },
228     /**
229     	 * Hides the control's close buttons
230     	 *
231     	 * @example $(element).monobombNavigator('hideClose')
232     	 *
233     	 * @returns {object} Jquery object
234     */
235 
236     hideClose: function() {
237       this.data('monobombNavigator').$close.hide();
238       return this;
239     },
240     /**
241     	 * Shows the control's close buttons
242     	 *
243     	 * @example $(element).monobombNavigator('showClose')
244     	 *
245     	 * @returns {object} Jquery object
246     */
247 
248     showClose: function() {
249       this.data('monobombNavigator').$close.show();
250       return this;
251     },
252     /**
253     	 * @example $(element).monobombNavigator('isAnimating')
254     	 *
255     	 * @returns {boolean} Whether the nav is animating
256     */
257 
258     isAnimating: function() {
259       return this.data('monobombNavigator').$inner_nav_wrapper.is(':animated');
260     },
261     /**
262     	 * @example $(element).monobombNavigator('isOpen')
263     	 *
264     	 * @returns {boolean} of whether the nav is open
265     */
266 
267     isOpen: function() {
268       return this.data('monobombNavigator').$main_nav_wrapper.position().left === 0;
269     },
270     /**
271     	 * Initializes the navigator
272     	 *
273     	 * @example $(element).monobombNavigator({
274     	 *              main_nav_wrapper: ".members",
275     	 *              inner_nav_wrapper: ".collaborator-wrapper",
276     	 *              inner_elements: "dl",
277     	 *              controls_nav: ".pager",
278     	 *              visible_columns: 2
279     	 *          })
280     	 *
281     	 * @param {object} options Various settings for the navigator
282     	 * @config {string} main_nav_wrapper Outer wrapper selector
283     	 * @config {string} inner_nav_wrapper Inner wrapper selector
284     	 * @config {string} inner_elements Page selector
285     	 * @config {string} controls_nav Control selector
286     	 * @config {string} [more_selector] More button selector ('> .more')
287     	 * @config {string} [back_selector] Back button selector '.back')
288     	 * @config {string} [close_selector] Close button selector ('.close')
289     	 * @config {string} [forward_selector] Forward button selector ('.forward')
290     	 * @config {string} [visible_columns] Number of columns visible at once (4)
291     	 *
292     	 * @returns {object} Jquery object
293     	 * @constructs
294     */
295 
296     init: function(options) {
297       var $first_page, data, individual_width, required_height, settings,
298         _this = this;
299       settings = {
300         more_selector: '> .more',
301         back_selector: '.back',
302         close_selector: '.close',
303         forward_selector: '.forward',
304         visible_columns: 4,
305         main_nav_wrapper: null,
306         inner_nav_wrapper: null,
307         inner_elements: null,
308         controls_nav: null
309       };
310       settings = $.extend(settings, options);
311       if (!(settings.main_nav_wrapper != null) || !(settings.inner_nav_wrapper != null) || !(settings.controls_nav != null)) {
312         console.log('[monobombNavigator] Error: required settings not defined.');
313         return false;
314       }
315       data = {
316         settings: settings
317       };
318       data.$main_nav_wrapper = $(settings.main_nav_wrapper);
319       data.$inner_nav_wrapper = data.$main_nav_wrapper.find(settings.inner_nav_wrapper);
320       data.$controls_nav = $(settings.controls_nav);
321       data.$more = this.find(settings.more_selector);
322       data.$back = data.$controls_nav.find(settings.back_selector);
323       data.$close = data.$controls_nav.find(settings.close_selector);
324       data.$forward = data.$controls_nav.find(settings.forward_selector);
325       data.$actual_navs = data.$inner_nav_wrapper.find(settings.inner_elements);
326       data.nav_count = data.$actual_navs.length;
327       data.original_left = 960 - $(data.$actual_navs[1]).width();
328       data.viewing = 1;
329       data.first_page = 1;
330       data.closed = !data.$main_nav_wrapper.hasClass('start-open');
331       this.data('monobombNavigator', data);
332       individual_width = $(data.$actual_navs[1]).outerWidth(true);
333       data.$inner_nav_wrapper.width(data.nav_count * individual_width + 'px');
334       required_height = data.$inner_nav_wrapper.height();
335       if (this.height() < required_height) {
336         this.height(required_height + 140);
337       }
338       if (data.closed && ($first_page = this.find('.first-page')).length) {
339         methods.moveToPage.call(this, $first_page, true);
340       }
341       methods.setButtons.apply(this);
342       data.$more.click(function() {
343         methods.openToPage.apply(_this);
344         return false;
345       });
346       data.$close.click(function() {
347         methods.closeToPage.apply(_this);
348         return false;
349       });
350       data.$forward.click(function() {
351         methods.forward.apply(_this);
352         return false;
353       });
354       data.$back.click(function() {
355         methods.back.apply(_this);
356         return false;
357       });
358       return this;
359     }
360   };

367     /**
368     	 * Returns the numberof the specified page
369     	 *
370     	 * @private
371     	 * @param {mixed} [page] Page number (1 indexed) or the actual page itself
372     	 *
373     	 * @returns {int} Page number
374     */
375 
376     getPageNumber: function(page) {
377       var $page, data, nav, page_num, _i, _len, _ref;
378       data = this.data('monobombNavigator');
379       if (typeof page === 'object') {
380         $page = $(page);
381         page_num = 1;
382         _ref = data.$actual_navs;
383         for (_i = 0, _len = _ref.length; _i < _len; _i++) {
384           nav = _ref[_i];
385           if ($page.equals(nav)) {
386             break;
387           }
388           page_num++;
389         }
390         return page_num;
391       }
392       return parseInt(page);
393     },
394     /**
395     	 * Returns the jQuery object of the specified page
396     	 *
397     	 * @private
398     	 * @param {mixed} [page] Page number (1 indexed) or the actual page itself
399     	 *
400     	 * @returns {object} Jquery object
401     */
402 
403     getPageObject: function(page) {
404       var data;
405       if (typeof page === 'object') {
406         return $(page);
407       }
408       data = this.data('monobombNavigator');
409       return data.$inner_nav_wrapper.find(data.settings.inner_elements + ':nth-child(' + parseInt(page) + ')');
410     }
411   };

420   $.fn.monobombNavigator = function(method) {
421     if (methods[method]) {
422       return methods[method].apply(this, Array.prototype.slice.call(arguments, 1));
423     }
424     return methods.init.apply(this, arguments);
